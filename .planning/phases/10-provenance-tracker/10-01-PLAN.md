  <action>
    Replace the entire contents of `tracker.py` with the following implementation.

    Key design points:
    - ProvenanceNode: frozen Pydantic v2 model, created_at always UTC
    - BoundaryCrossing: frozen Pydantic v2 model replacing dict return type
    - ProvenanceTracker maintains _nodes dict AND _children reverse-index
    - _would_create_cycle: iterative DFS walking parent links upward; returns True if new_id is reachable from any proposed parent
    - add_artifact: validates duplicate, unknown parent, cycle — all BEFORE any mutation
    - trace_backward / trace_forward: iterative DFS using explicit stack (no recursion)
    - get_isolation_boundary_crossings: walks full ancestor subgraph via trace_backward, checks every edge, deduplicates with seen_edges set
    - to_json: topological sort (Kahn's algorithm) then json.dumps
    - from_json: parse JSON, call add_artifact in list order, raise DeserializationError on bad JSON, CycleDetectedError on cyclic data
    - get_node: direct dict lookup, returns None if missing
    - get_by_type: linear scan, exact case-sensitive match

    ```python
    from __future__ import annotations

    import json
    from collections import deque
    from datetime import UTC, datetime
    from typing import Any

    from pydantic import BaseModel, Field

    from .exceptions import (
        CycleDetectedError,
        DeserializationError,
        DuplicateArtifactError,
        UnknownParentError,
    )


    class ProvenanceNode(BaseModel, frozen=True):
        """Immutable graph node representing one tracked artifact."""

        artifact_id: str
        artifact_type: str
        metadata: dict[str, Any] = Field(default_factory=dict)
        parents: list[str] = Field(default_factory=list)
        created_at: datetime = Field(default_factory=lambda: datetime.now(UTC))


    class BoundaryCrossing(BaseModel, frozen=True):
        """A detected isolation boundary crossing between two nodes."""

        from_node: str
        to_node: str
        from_boundary: Any  # value of boundary key on parent (None if absent)
        to_boundary: Any    # value of boundary key on child (None if absent)
        boundary_key: str


    class ProvenanceTracker:
        """Forensic data lineage tracker — maintains a DAG of artifacts."""

        def __init__(self) -> None:
            self._nodes: dict[str, ProvenanceNode] = {}
            self._children: dict[str, set[str]] = {}

        # ------------------------------------------------------------------
        # Mutation
        # ------------------------------------------------------------------

        def add_artifact(
            self,
            artifact_id: str,
            artifact_type: str,
            metadata: dict[str, Any] | None = None,
            parents: list[str] | None = None,
        ) -> ProvenanceNode:
            """Register a new artifact. Raises before any mutation on error."""
            parents_list = list(parents) if parents else []

            if artifact_id in self._nodes:
                raise DuplicateArtifactError(
                    f"Artifact '{artifact_id}' already exists in the graph."
                )
            for pid in parents_list:
                if pid not in self._nodes:
                    raise UnknownParentError(
                        f"Parent '{pid}' is not present in the graph."
                    )
            if self._would_create_cycle(artifact_id, parents_list):
                raise CycleDetectedError(
                    f"Adding '{artifact_id}' with parents {parents_list} would create a cycle."
                )

            node = ProvenanceNode(
                artifact_id=artifact_id,
                artifact_type=artifact_type,
                metadata=metadata or {},
                parents=parents_list,
            )
            self._nodes[artifact_id] = node
            self._children[artifact_id] = set()
            for pid in parents_list:
                self._children[pid].add(artifact_id)
            return node

        def _would_create_cycle(self, new_id: str, parents: list[str]) -> bool:
            """Return True if new_id is already reachable from any node in parents."""
            visited: set[str] = set()
            stack = list(parents)
            while stack:
                curr = stack.pop()
                if curr == new_id:
                    return True
                if curr in visited:
                    continue
                visited.add(curr)
                if curr in self._nodes:
                    stack.extend(self._nodes[curr].parents)
            return False

        # ------------------------------------------------------------------
        # Queries
        # ------------------------------------------------------------------

        def get_node(self, artifact_id: str) -> ProvenanceNode | None:
            """Return the node for artifact_id, or None if not present."""
            return self._nodes.get(artifact_id)

        def trace_backward(self, artifact_id: str) -> list[ProvenanceNode]:
            """Return all ancestors of artifact_id (inclusive), no duplicates."""
            if artifact_id not in self._nodes:
                return []
            result: list[ProvenanceNode] = []
            visited: set[str] = set()
            stack = [artifact_id]
            while stack:
                curr = stack.pop()
                if curr in visited or curr not in self._nodes:
                    continue
                visited.add(curr)
                result.append(self._nodes[curr])
                stack.extend(self._nodes[curr].parents)
            return result

        def trace_forward(self, artifact_id: str) -> list[ProvenanceNode]:
            """Return all descendants of artifact_id (inclusive), no duplicates."""
            if artifact_id not in self._nodes:
                return []
            result: list[ProvenanceNode] = []
            visited: set[str] = set()
            stack = [artifact_id]
            while stack:
                curr = stack.pop()
                if curr in visited or curr not in self._nodes:
                    continue
                visited.add(curr)
                result.append(self._nodes[curr])
                stack.extend(self._children.get(curr, set()))
            return result

        def get_isolation_boundary_crossings(
            self, artifact_id: str, boundary: str
        ) -> list[BoundaryCrossing]:
            """Return all boundary crossings in the full ancestor subgraph."""
            ancestors = self.trace_backward(artifact_id)
            if not ancestors:
                return []
            ancestor_ids = {n.artifact_id for n in ancestors}
            crossings: list[BoundaryCrossing] = []
            seen_edges: set[tuple[str, str]] = set()
            for node in ancestors:
                for pid in node.parents:
                    if pid not in ancestor_ids:
                        continue
                    edge = (pid, node.artifact_id)
                    if edge in seen_edges:
                        continue
                    seen_edges.add(edge)
                    parent = self._nodes[pid]
                    pv = parent.metadata.get(boundary)
                    cv = node.metadata.get(boundary)
                    if pv != cv:
                        crossings.append(
                            BoundaryCrossing(
                                from_node=pid,
                                to_node=node.artifact_id,
                                from_boundary=pv,
                                to_boundary=cv,
                                boundary_key=boundary,
                            )
                        )
            return crossings

        def get_by_type(self, artifact_type: str) -> list[ProvenanceNode]:
            """Return all nodes whose artifact_type exactly matches (case-sensitive)."""
            return [n for n in self._nodes.values() if n.artifact_type == artifact_type]

        # ------------------------------------------------------------------
        # Serialization
        # ------------------------------------------------------------------

        def to_json(self) -> str:
            """Serialize the graph to JSON with nodes in topological order."""
            ordered = self._topological_sort()
            nodes_data = []
            for node in ordered:
                nodes_data.append({
                    "artifact_id": node.artifact_id,
                    "artifact_type": node.artifact_type,
                    "metadata": node.metadata,
                    "parents": node.parents,
                    "created_at": node.created_at.isoformat(),
                })
            return json.dumps({"nodes": nodes_data})

        @classmethod
        def from_json(cls, data: str) -> ProvenanceTracker:
            """Reconstruct a ProvenanceTracker from a JSON string produced by to_json."""
            try:
                parsed = json.loads(data)
                nodes_data = parsed["nodes"]
            except (json.JSONDecodeError, KeyError, TypeError) as exc:
                raise DeserializationError(f"Malformed provenance JSON: {exc}") from exc

            tracker = cls()
            for entry in nodes_data:
                try:
                    created_at = datetime.fromisoformat(entry["created_at"])
                    tracker.add_artifact(
                        artifact_id=entry["artifact_id"],
                        artifact_type=entry["artifact_type"],
                        metadata=entry.get("metadata", {}),
                        parents=entry.get("parents", []),
                    )
                    # Overwrite the node to restore the original created_at timestamp
                    node = tracker._nodes[entry["artifact_id"]]
                    tracker._nodes[entry["artifact_id"]] = ProvenanceNode(
                        artifact_id=node.artifact_id,
                        artifact_type=node.artifact_type,
                        metadata=node.metadata,
                        parents=node.parents,
                        created_at=created_at,
                    )
                except (DuplicateArtifactError, UnknownParentError, CycleDetectedError):
                    raise
                except (KeyError, TypeError, ValueError) as exc:
                    raise DeserializationError(f"Invalid node entry: {exc}") from exc
            return tracker

        def _topological_sort(self) -> list[ProvenanceNode]:
            """Return nodes in topological order (parents before children) via Kahn's algorithm."""
            in_degree: dict[str, int] = {nid: 0 for nid in self._nodes}
            for node in self._nodes.values():
                for pid in node.parents:
                    in_degree[node.artifact_id] = in_degree.get(node.artifact_id, 0) + 1

            # Recompute properly
            in_degree = {nid: len(self._nodes[nid].parents) for nid in self._nodes}
            queue: deque[str] = deque(nid for nid, deg in in_degree.items() if deg == 0)
            result: list[ProvenanceNode] = []
            while queue:
                curr = queue.popleft()
                result.append(self._nodes[curr])
                for child_id in self._children.get(curr, set()):
                    in_degree[child_id] -= 1
                    if in_degree[child_id] == 0:
                        queue.append(child_id)
            return result
    ```
