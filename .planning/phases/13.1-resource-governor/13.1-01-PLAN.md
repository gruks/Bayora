<objective>
Extend the existing resource governor stub into a production-ready cgroup v2 resource control component.

The stub already has: ResourceQuotaSpec (plain Pydantic model), ResourceGovernor with generate_cgroup_spec and validate_quota.

This plan adds:

1. ResourceQuotaSpec: upgrade to frozen=True, add cpu_pin (list of CPU core indices to pin to)
2. TenantResourceUsage: new frozen Pydantic v2 model — cpu_percent, memory_bytes, io_bytes_per_sec, pid_count, tenant_id, timestamp
3. ResourceViolation: new frozen Pydantic v2 model — tenant_id, resource_type, current_value, limit_value, timestamp
4. apply_timing_jitter: sleeps a random duration between 50ms–200ms (prevents timing side-channel attacks), returns actual sleep duration in ms
5. record_usage: accepts a TenantResourceUsage, emits Prometheus Gauge metrics per tenant per dimension
6. check_violations: compares a TenantResourceUsage against the quota, returns list[ResourceViolation] for any exceeded limit
7. generate_kubernetes_resources: produces a K8s ResourceRequirements dict (requests + limits) from the quota spec
8. Unit tests covering all new capabilities

  <action>
    Replace the entire contents of `governor.py` with the following implementation.

    Key design points:
    - ResourceQuotaSpec: frozen=True, add cpu_pin field
    - TenantResourceUsage: new frozen model with four resource dimensions + tenant_id + timestamp
    - ResourceViolation: new frozen model for a single exceeded limit
    - apply_timing_jitter: random.uniform(0.05, 0.20) sleep, returns actual ms slept; accepts optional `_rng` for test determinism
    - record_usage: updates Prometheus Gauges labeled by tenant_id and resource_type
    - check_violations: compares usage against quota, returns one ResourceViolation per exceeded dimension
    - generate_cgroup_spec: keep existing logic, add cpu_pin → cpuset.cpus entry
    - generate_kubernetes_resources: converts quota to K8s resource requests/limits dict
    - validate_quota: keep existing logic

    ```python
    from __future__ import annotations

    import random
    import time
    from datetime import UTC, datetime
    from typing import Any

    from prometheus_client import Counter, Gauge
    from pydantic import BaseModel, Field

    # ---------------------------------------------------------------------------
    # Prometheus metrics
    # ---------------------------------------------------------------------------

    _resource_limit_gauge = Gauge(
        "centinela_resource_limit",
        "Configured resource limit per tenant",
        ["tenant_id", "resource_type"],
    )
    _resource_usage_gauge = Gauge(
        "centinela_resource_usage",
        "Current resource usage per tenant",
        ["tenant_id", "resource_type"],
    )
    _resource_violations_counter = Counter(
        "centinela_resource_violations_total",
        "Total resource limit violations per tenant",
        ["tenant_id", "resource_type"],
    )
    _timing_jitter_gauge = Gauge(
        "centinela_timing_jitter_ms",
        "Last applied timing jitter in milliseconds",
        ["tenant_id"],
    )

    # ---------------------------------------------------------------------------
    # Data models
    # ---------------------------------------------------------------------------

    class ResourceQuotaSpec(BaseModel, frozen=True):
        """Resource quota specification for a cgroup v2 tenant."""

        cpu_quota: int | None = None        # microseconds per cpu_period
        cpu_period: int = 100_000           # default 100ms in microseconds
        cpu_pin: list[int] = Field(default_factory=list)  # CPU core indices to pin
        memory_high: int | None = None      # soft limit in bytes
        memory_max: int | None = None       # hard limit in bytes (OOM killer threshold)
        io_weight: int | None = Field(default=None, ge=1, le=10_000)
        pids_max: int | None = None


    class TenantResourceUsage(BaseModel, frozen=True):
        """Observed resource usage snapshot for a tenant."""

        tenant_id: str
        cpu_percent: float          # 0.0–100.0 per core
        memory_bytes: int           # current RSS in bytes
        io_bytes_per_sec: float     # combined read+write bytes/sec
        pid_count: int              # number of active PIDs
        timestamp: datetime = Field(default_factory=lambda: datetime.now(UTC))


    class ResourceViolation(BaseModel, frozen=True):
        """A single resource limit violation."""

        tenant_id: str
        resource_type: str          # "cpu", "memory", "io", "pids"
        current_value: float
        limit_value: float
        timestamp: datetime = Field(default_factory=lambda: datetime.now(UTC))

    # ---------------------------------------------------------------------------
    # ResourceGovernor
    # ---------------------------------------------------------------------------

    # Timing jitter bounds (seconds)
    _JITTER_MIN_S = 0.050   # 50ms
    _JITTER_MAX_S = 0.200   # 200ms


    class ResourceGovernor:
        """Enforces cgroup v2 resource limits with timing jitter and Prometheus metrics."""

        def __init__(self, quota: ResourceQuotaSpec) -> None:
            self.quota = quota

        # ------------------------------------------------------------------
        # cgroup v2 spec generation
        # ------------------------------------------------------------------

        def generate_cgroup_spec(self) -> dict[str, str]:
            """Produce a cgroup v2 settings dict suitable for writing to cgroup files."""
            spec: dict[str, str] = {}

            if self.quota.cpu_quota is not None:
                spec["cpu.max"] = f"{self.quota.cpu_quota} {self.quota.cpu_period}"

            if self.quota.cpu_pin:
                # e.g. [0, 1, 2] → "0-2" or [0, 2] → "0,2"
                spec["cpuset.cpus"] = _format_cpu_list(self.quota.cpu_pin)

            if self.quota.memory_high is not None:
                spec["memory.high"] = str(self.quota.memory_high)

            if self.quota.memory_max is not None:
                spec["memory.max"] = str(self.quota.memory_max)

            if self.quota.io_weight is not None:
                spec["io.weight"] = str(self.quota.io_weight)

            if self.quota.pids_max is not None:
                spec["pids.max"] = str(self.quota.pids_max)

            return spec

        # ------------------------------------------------------------------
        # Kubernetes resource spec generation
        # ------------------------------------------------------------------

        def generate_kubernetes_resources(self) -> dict[str, Any]:
            """Produce a Kubernetes ResourceRequirements dict.

            Converts cgroup v2 quota values to K8s CPU (millicores) and memory (bytes).
            """
            requests: dict[str, str] = {}
            limits: dict[str, str] = {}

            if self.quota.cpu_quota is not None and self.quota.cpu_period > 0:
                # cpu_quota / cpu_period = fraction of one CPU core
                cpu_fraction = self.quota.cpu_quota / self.quota.cpu_period
                millicores = int(cpu_fraction * 1000)
                requests["cpu"] = f"{millicores}m"
                limits["cpu"] = f"{millicores}m"

            if self.quota.memory_max is not None:
                limits["memory"] = str(self.quota.memory_max)
            if self.quota.memory_high is not None:
                requests["memory"] = str(self.quota.memory_high)

            return {"requests": requests, "limits": limits}

        # ------------------------------------------------------------------
        # Timing jitter (side-channel mitigation)
        # ------------------------------------------------------------------

        def apply_timing_jitter(
            self,
            tenant_id: str = "default",
            _rng: random.Random | None = None,
        ) -> float:
            """Sleep a random duration between 50ms and 200ms.

            Prevents timing side-channel attacks by introducing unpredictable
            response latency. Returns the actual sleep duration in milliseconds.

            Args:
                tenant_id: Used for Prometheus metric labeling.
                _rng: Optional Random instance for deterministic testing.

            Returns:
                Actual sleep duration in milliseconds.
            """
            rng = _rng or random
            jitter_s = rng.uniform(_JITTER_MIN_S, _JITTER_MAX_S)
            time.sleep(jitter_s)
            jitter_ms = jitter_s * 1000
            _timing_jitter_gauge.labels(tenant_id=tenant_id).set(jitter_ms)
            return jitter_ms

        # ------------------------------------------------------------------
        # Usage tracking and violation detection
        # ------------------------------------------------------------------

        def record_usage(self, usage: TenantResourceUsage) -> None:
            """Emit Prometheus Gauge metrics for all four resource dimensions."""
            tid = usage.tenant_id
            _resource_usage_gauge.labels(tenant_id=tid, resource_type="cpu").set(
                usage.cpu_percent
            )
            _resource_usage_gauge.labels(tenant_id=tid, resource_type="memory").set(
                usage.memory_bytes
            )
            _resource_usage_gauge.labels(tenant_id=tid, resource_type="io").set(
                usage.io_bytes_per_sec
            )
            _resource_usage_gauge.labels(tenant_id=tid, resource_type="pids").set(
                usage.pid_count
            )

        def check_violations(
            self, usage: TenantResourceUsage
        ) -> list[ResourceViolation]:
            """Compare usage against quota and return violations for exceeded limits.

            Checks:
            - cpu_percent > (cpu_quota / cpu_period) * 100  (if cpu_quota set)
            - memory_bytes > memory_max                      (if memory_max set)
            - io_bytes_per_sec > io_weight * 1024            (if io_weight set, rough proxy)
            - pid_count > pids_max                           (if pids_max set)
            """
            violations: list[ResourceViolation] = []
            tid = usage.tenant_id

            # CPU
            if self.quota.cpu_quota is not None and self.quota.cpu_period > 0:
                cpu_limit_pct = (self.quota.cpu_quota / self.quota.cpu_period) * 100
                if usage.cpu_percent > cpu_limit_pct:
                    violations.append(ResourceViolation(
                        tenant_id=tid,
                        resource_type="cpu",
                        current_value=usage.cpu_percent,
                        limit_value=cpu_limit_pct,
                    ))
                    _resource_violations_counter.labels(
                        tenant_id=tid, resource_type="cpu"
                    ).inc()

            # Memory
            if self.quota.memory_max is not None:
                if usage.memory_bytes > self.quota.memory_max:
                    violations.append(ResourceViolation(
                        tenant_id=tid,
                        resource_type="memory",
                        current_value=float(usage.memory_bytes),
                        limit_value=float(self.quota.memory_max),
                    ))
                    _resource_violations_counter.labels(
                        tenant_id=tid, resource_type="memory"
                    ).inc()

            # I/O (io_weight * 1024 bytes/sec as a rough proxy limit)
            if self.quota.io_weight is not None:
                io_limit = float(self.quota.io_weight) * 1024
                if usage.io_bytes_per_sec > io_limit:
                    violations.append(ResourceViolation(
                        tenant_id=tid,
                        resource_type="io",
                        current_value=usage.io_bytes_per_sec,
                        limit_value=io_limit,
                    ))
                    _resource_violations_counter.labels(
                        tenant_id=tid, resource_type="io"
                    ).inc()

            # PIDs
            if self.quota.pids_max is not None:
                if usage.pid_count > self.quota.pids_max:
                    violations.append(ResourceViolation(
                        tenant_id=tid,
                        resource_type="pids",
                        current_value=float(usage.pid_count),
                        limit_value=float(self.quota.pids_max),
                    ))
                    _resource_violations_counter.labels(
                        tenant_id=tid, resource_type="pids"
                    ).inc()

            return violations

        # ------------------------------------------------------------------
        # Validation
        # ------------------------------------------------------------------

        def validate_quota(self) -> list[str]:
            """Return a list of validation errors for the quota spec."""
            errors: list[str] = []
            if self.quota.cpu_quota is not None and self.quota.cpu_quota <= 0:
                errors.append("cpu_quota must be positive")
            if self.quota.cpu_period <= 0:
                errors.append("cpu_period must be positive")
            if (
                self.quota.memory_max is not None
                and self.quota.memory_high is not None
                and self.quota.memory_high > self.quota.memory_max
            ):
                errors.append("memory_high cannot exceed memory_max")
            for core in self.quota.cpu_pin:
                if core < 0:
                    errors.append(f"cpu_pin core index {core} must be non-negative")
            return errors


    # ---------------------------------------------------------------------------
    # Helpers
    # ---------------------------------------------------------------------------

    def _format_cpu_list(cores: list[int]) -> str:
        """Format a list of CPU core indices as a cpuset string.

    Examples:
        [0, 1, 2] → "0-2"
        [0, 2, 4] → "0,2,4"
        [0, 1, 3] → "0-1,3"
    """
        if not cores:
            return ""
        sorted_cores = sorted(set(cores))
        ranges: list[str] = []
        start = sorted_cores[0]
        end = sorted_cores[0]
        for c in sorted_cores[1:]:
            if c == end + 1:
                end = c
            else:
                ranges.append(f"{start}-{end}" if start != end else str(start))
                start = end = c
        ranges.append(f"{start}-{end}" if start != end else str(start))
        return ",".join(ranges)
    ```
