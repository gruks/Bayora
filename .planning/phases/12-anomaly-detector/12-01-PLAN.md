    Replace the entire contents of `detector.py` with the following implementation.

    Key design points:
    - AlertType: StrEnum with four values
    - AnomalyAlert: frozen Pydantic v2 model — replaces dict return type
    - WelfordStat: keep existing implementation unchanged (it works correctly)
    - AnomalyDetector: extend with four specialised detect_* methods and a unified observe_and_detect
    - Network exfiltration: threshold is 10_485_760 bytes/min (10 MB/min)
    - Container escape: any observation of metric_name="escape_attempt" with value > 0 is immediately critical
    - Timing side-channel: uses the existing statistical baseline on "response_time_ms" metric
    - All alerts emit Prometheus metrics

    ```python
    from __future__ import annotations

    import math
    from datetime import UTC, datetime
    from enum import StrEnum
    from typing import Any

    from prometheus_client import Counter, Gauge
    from pydantic import BaseModel, Field

    # ---------------------------------------------------------------------------
    # Prometheus metrics
    # ---------------------------------------------------------------------------

    _metric_value_gauge = Gauge(
        "centinela_metric_value",
        "Current value of a monitored metric",
        ["metric_name"],
    )
    _anomaly_counter = Counter(
        "centinela_anomaly_total",
        "Total number of detected anomalies",
        ["alert_type", "severity"],
    )

    # ---------------------------------------------------------------------------
    # Types
    # ---------------------------------------------------------------------------

    class AlertType(StrEnum):
        STATISTICAL = "statistical"
        NETWORK_EXFILTRATION = "network_exfiltration"
        CONTAINER_ESCAPE = "container_escape"
        TIMING_SIDE_CHANNEL = "timing_side_channel"


    class AnomalyAlert(BaseModel, frozen=True):
        """Immutable alert produced when an anomaly is detected."""

        metric_name: str
        value: float
        expected_mean: float
        std_dev: float
        z_score: float
        severity: str          # "medium" | "high" | "critical"
        alert_type: AlertType
        timestamp: datetime = Field(default_factory=lambda: datetime.now(UTC))

    # ---------------------------------------------------------------------------
    # Welford sliding-window statistics (unchanged from stub)
    # ---------------------------------------------------------------------------

    class WelfordStat:
        """Running statistics using Welford's algorithm over a sliding window."""

        def __init__(self, window_size: int) -> None:
            self.window_size = window_size
            self.count = 0
            self.mean = 0.0
            self.m2 = 0.0
            self._min = float("inf")
            self._max = float("-inf")
            self.values: list[float] = []

        def update(self, value: float) -> None:
            """Update running statistics with a new observation."""
            self.values.append(value)
            if len(self.values) > self.window_size:
                self.values.pop(0)
                self.count = len(self.values)
                self.mean = sum(self.values) / self.count
                self.m2 = sum((x - self.mean) ** 2 for x in self.values)
                self._min = min(self.values)
                self._max = max(self.values)
                return

            self.count += 1
            delta = value - self.mean
            self.mean += delta / self.count
            delta2 = value - self.mean
            self.m2 += delta * delta2

            if value < self._min:
                self._min = value
            if value > self._max:
                self._max = value

        @property
        def variance(self) -> float:
            if self.count < 2:
                return 0.0
            return self.m2 / (self.count - 1)

        @property
        def std_dev(self) -> float:
            return math.sqrt(self.variance)

    # ---------------------------------------------------------------------------
    # AnomalyDetector
    # ---------------------------------------------------------------------------

    # Network exfiltration threshold: 10 MB/min
    NETWORK_EXFILTRATION_THRESHOLD_BYTES_PER_MIN: float = 10 * 1024 * 1024  # 10_485_760

    class AnomalyDetector:
        """Real-time security anomaly detector.

        Supports four detection categories:
        - STATISTICAL: generic metric deviates >threshold_sigma from baseline
        - NETWORK_EXFILTRATION: bytes_per_minute exceeds 10 MB/min
        - CONTAINER_ESCAPE: any escape_attempt observation with value > 0
        - TIMING_SIDE_CHANNEL: response_time_ms deviates >threshold_sigma from baseline
        """

        def __init__(
            self,
            window_size: int = 100,
            threshold_sigma: float = 3.0,
        ) -> None:
            self.window_size = window_size
            self.threshold_sigma = threshold_sigma
            self._stats: dict[str, WelfordStat] = {}
            self._latest_values: dict[str, float] = {}

        # ------------------------------------------------------------------
        # Core observation
        # ------------------------------------------------------------------

        def observe(self, metric_name: str, value: float) -> None:
            """Record a new metric observation and update the baseline."""
            if metric_name not in self._stats:
                self._stats[metric_name] = WelfordStat(self.window_size)
            self._stats[metric_name].update(value)
            self._latest_values[metric_name] = value
            _metric_value_gauge.labels(metric_name=metric_name).set(value)

        def observe_and_detect(
            self, metric_name: str, value: float
        ) -> list[AnomalyAlert]:
            """Observe a value and immediately return any triggered alerts.

            Dispatches to the appropriate specialised detector based on metric_name:
            - "escape_attempt"   → detect_container_escape
            - "bytes_per_minute" → detect_network_exfiltration
            - "response_time_ms" → detect_timing_side_channel
            - anything else      → detect_statistical
            """
            self.observe(metric_name, value)

            if metric_name == "escape_attempt":
                return self.detect_container_escape(value)
            if metric_name == "bytes_per_minute":
                return self.detect_network_exfiltration(value)
            if metric_name == "response_time_ms":
                return self.detect_timing_side_channel()
            return self.detect_statistical(metric_name)

        # ------------------------------------------------------------------
        # Specialised detectors
        # ------------------------------------------------------------------

        def detect_statistical(self, metric_name: str) -> list[AnomalyAlert]:
            """Generic statistical anomaly: latest value deviates >threshold_sigma."""
            if metric_name not in self._stats or metric_name not in self._latest_values:
                return []

            stat = self._stats[metric_name]
            if stat.count < 2:
                return []

            latest = self._latest_values[metric_name]
            std = stat.std_dev
            if std <= 0:
                return []

            z_score = abs(latest - stat.mean) / std
            if z_score <= self.threshold_sigma:
                return []

            severity = "high" if z_score > self.threshold_sigma * 1.5 else "medium"
            alert = AnomalyAlert(
                metric_name=metric_name,
                value=latest,
                expected_mean=stat.mean,
                std_dev=std,
                z_score=z_score,
                severity=severity,
                alert_type=AlertType.STATISTICAL,
            )
            _anomaly_counter.labels(
                alert_type=AlertType.STATISTICAL, severity=severity
            ).inc()
            return [alert]

        def detect_network_exfiltration(
            self, bytes_per_minute: float
        ) -> list[AnomalyAlert]:
            """Fire if bytes_per_minute exceeds 10 MB/min threshold."""
            if bytes_per_minute <= NETWORK_EXFILTRATION_THRESHOLD_BYTES_PER_MIN:
                return []

            stat = self._stats.get("bytes_per_minute")
            mean = stat.mean if stat and stat.count >= 2 else 0.0
            std = stat.std_dev if stat and stat.count >= 2 else 0.0
            z_score = (
                abs(bytes_per_minute - mean) / std if std > 0 else float("inf")
            )
            severity = "critical" if bytes_per_minute > NETWORK_EXFILTRATION_THRESHOLD_BYTES_PER_MIN * 2 else "high"

            alert = AnomalyAlert(
                metric_name="bytes_per_minute",
                value=bytes_per_minute,
                expected_mean=mean,
                std_dev=std,
                z_score=z_score,
                severity=severity,
                alert_type=AlertType.NETWORK_EXFILTRATION,
            )
            _anomaly_counter.labels(
                alert_type=AlertType.NETWORK_EXFILTRATION, severity=severity
            ).inc()
            return [alert]

        def detect_container_escape(self, value: float) -> list[AnomalyAlert]:
            """Fire immediately (severity=critical) when escape_attempt value > 0."""
            if value <= 0:
                return []

            alert = AnomalyAlert(
                metric_name="escape_attempt",
                value=value,
                expected_mean=0.0,
                std_dev=0.0,
                z_score=float("inf"),
                severity="critical",
                alert_type=AlertType.CONTAINER_ESCAPE,
            )
            _anomaly_counter.labels(
                alert_type=AlertType.CONTAINER_ESCAPE, severity="critical"
            ).inc()
            return [alert]

        def detect_timing_side_channel(self) -> list[AnomalyAlert]:
            """Fire when response_time_ms deviates >threshold_sigma from baseline."""
            return self.detect_statistical("response_time_ms")

        # ------------------------------------------------------------------
        # Legacy interface (kept for backward compatibility)
        # ------------------------------------------------------------------

        def detect(self, metric_name: str) -> list[dict[str, Any]]:
            """Legacy dict-based detect. Prefer detect_statistical for new code."""
            alerts = self.detect_statistical(metric_name)
            return [
                {
                    "metric_name": a.metric_name,
                    "value": a.value,
                    "expected_mean": a.expected_mean,
                    "std_dev": a.std_dev,
                    "z_score": a.z_score,
                    "severity": a.severity,
                }
                for a in alerts
            ]

        def get_baseline(self, metric_name: str) -> dict[str, Any]:
            """Return mean, std, min, max for a metric."""
            if metric_name not in self._stats:
                return {"mean": 0.0, "std": 0.0, "min": 0.0, "max": 0.0}
            stat = self._stats[metric_name]
            return {
                "mean": stat.mean,
                "std": stat.std_dev,
                "min": stat._min if stat.count > 0 else 0.0,
                "max": stat._max if stat.count > 0 else 0.0,
            }
    ```
