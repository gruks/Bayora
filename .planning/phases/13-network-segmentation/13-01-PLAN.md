<objective>
Extend the existing NetworkManager stub into a production-ready WireGuard-based network segmentation component.

The stub already has: WireGuardConfig/PeerConfig Pydantic models, load_config (wgconfig parser), generate_keypair (wg subprocess with mock fallback), apply_policy (basic K8s NetworkPolicy dict), validate_config.

This plan adds:

1. WireGuardTunnel — frozen Pydantic v2 model representing a per-tenant tunnel (private_key, public_key, address, tenant_id)
2. NetworkBoundaryCrossing — frozen Pydantic v2 model for logging boundary crossing events
3. create_tenant_tunnel — generates a unique keypair and assigns a /32 address from a configurable subnet per tenant_id
4. generate_network_policy — typed replacement for apply_policy; returns a structured K8s NetworkPolicy dict with DNS tunneling prevention (port 53 blocked in DENY_ALL and WHITELIST)
5. add_egress_whitelist — adds specific CIDR/port rules to a WHITELIST policy
6. log_boundary_crossing — records when traffic crosses a tenant boundary
7. Unit tests covering all new capabilities

Output: 4 files — manager.py extended, __init__.py updated, tests/__init__.py, test_manager.py.
</objective>

    Key design points:

    - WireGuardTunnel: new frozen model — private_key, public_key, address (/32), tenant_id, created_at
    - NetworkBoundaryCrossing: new frozen model — tenant_id, direction ("ingress"|"egress"), remote_ip, port, timestamp
    - create_tenant_tunnel: generates keypair, assigns next available /32 from base_subnet, stores in _tunnels dict

    - add_egress_whitelist: adds a CIDR to the egress allow-list of an existing WHITELIST policy
    - log_boundary_crossing: appends a NetworkBoundaryCrossing to an in-memory log
    - get_boundary_crossings: returns all logged crossings, optionally filtered by tenant_id
    - validate_config: keep existing logic, upgrade return to list[str]
    - Legacy apply_policy: keep for backward compatibility, delegates to generate_network_policy

    ```python
    from __future__ import annotations

    import ipaddress
    import subprocess
    from datetime import UTC, datetime
    from typing import Any

    import wgconfig  # type: ignore[import-untyped]
    from pydantic import BaseModel, Field

    from centinela.enums import NetworkPolicyType

    # ---------------------------------------------------------------------------
    # Data models
    # ---------------------------------------------------------------------------

    class PeerConfig(BaseModel, frozen=True):
        """WireGuard peer configuration."""

        public_key: str
        allowed_ips: list[str]
        endpoint: str | None = None
        persistent_keepalive: int | None = None


    class WireGuardConfig(BaseModel, frozen=True):
        """WireGuard interface configuration."""

        interface_name: str
        private_key: str
        address: str
        dns: list[str] = Field(default_factory=list)
        peers: list[PeerConfig] = Field(default_factory=list)


    class WireGuardTunnel(BaseModel, frozen=True):
        """A per-tenant WireGuard tunnel with its own keypair and address."""

        tenant_id: str
        private_key: str
        public_key: str
        address: str          # e.g. "10.100.0.5/32"
        created_at: datetime = Field(default_factory=lambda: datetime.now(UTC))


    class NetworkBoundaryCrossing(BaseModel, frozen=True):
        """A logged network boundary crossing event."""

        tenant_id: str
        direction: str        # "ingress" or "egress"
        remote_ip: str
        port: int
        timestamp: datetime = Field(default_factory=lambda: datetime.now(UTC))


    # ---------------------------------------------------------------------------
    # DNS tunneling prevention — always block port 53
    # ---------------------------------------------------------------------------

    _DNS_BLOCK_EGRESS: dict[str, Any] = {
        "ports": [{"protocol": "UDP", "port": 53}, {"protocol": "TCP", "port": 53}],
        "to": [],  # empty to = block all destinations on these ports
    }

    # ---------------------------------------------------------------------------
    # NetworkManager
    # ---------------------------------------------------------------------------

    # Default subnet for tenant tunnel address allocation
    _DEFAULT_TUNNEL_SUBNET = "10.100.0.0/16"
    _TUNNEL_BASE_OFFSET = 2  # start at .2 (reserve .1 for gateway)


    class NetworkManager:
        """WireGuard-based network segmentation and policy manager."""

        def __init__(self, tunnel_subnet: str = _DEFAULT_TUNNEL_SUBNET) -> None:
            self._tunnel_subnet = ipaddress.IPv4Network(tunnel_subnet, strict=False)
            self._tunnels: dict[str, WireGuardTunnel] = {}
            self._next_host_index: int = _TUNNEL_BASE_OFFSET
            self._boundary_crossings: list[NetworkBoundaryCrossing] = []

        # ------------------------------------------------------------------
        # Tunnel management
        # ------------------------------------------------------------------

        def create_tenant_tunnel(self, tenant_id: str) -> WireGuardTunnel:
            """Create a WireGuard tunnel for a tenant with a unique keypair and /32 address.

            Raises ValueError if tenant_id already has a tunnel.
            """
            if tenant_id in self._tunnels:
                raise ValueError(f"Tunnel for tenant '{tenant_id}' already exists.")

            private_key, public_key = self.generate_keypair()

            hosts = list(self._tunnel_subnet.hosts())
            if self._next_host_index >= len(hosts):
                raise RuntimeError("Tunnel subnet exhausted — no more addresses available.")

            address = f"{hosts[self._next_host_index]}/32"
            self._next_host_index += 1

            tunnel = WireGuardTunnel(
                tenant_id=tenant_id,
                private_key=private_key,
                public_key=public_key,
                address=address,
            )
            self._tunnels[tenant_id] = tunnel
            return tunnel

        def get_tunnel(self, tenant_id: str) -> WireGuardTunnel | None:
            """Return the tunnel for a tenant, or None if not found."""
            return self._tunnels.get(tenant_id)

        # ------------------------------------------------------------------
        # Keypair generation
        # ------------------------------------------------------------------

        def generate_keypair(self) -> tuple[str, str]:
            """Generate a WireGuard private/public keypair.

            Falls back to deterministic mock strings when the `wg` binary is absent
            (CI environments, Windows, containers without wireguard-tools).
            """
            try:
                privkey_proc = subprocess.run(
                    ["wg", "genkey"], capture_output=True, text=True, check=True
                )
                privkey = privkey_proc.stdout.strip()
                pubkey_proc = subprocess.run(
                    ["wg", "pubkey"],
                    input=privkey,
                    capture_output=True,
                    text=True,
                    check=True,
                )
                pubkey = pubkey_proc.stdout.strip()
                return privkey, pubkey
            except (subprocess.CalledProcessError, FileNotFoundError):
                return "mock_priv_key_base64_string=", "mock_pub_key_base64_string="

        # ------------------------------------------------------------------
        # Network policy generation
        # ------------------------------------------------------------------

        def generate_network_policy(
            self,
            policy_type: NetworkPolicyType,
            tenant_id: str,
            egress_whitelist: list[str] | None = None,
        ) -> dict[str, Any]:
            """Generate a Kubernetes NetworkPolicy manifest.

            DNS tunneling prevention: port 53 (UDP and TCP) is always blocked
            in DENY_ALL and WHITELIST policies.

            Args:
                policy_type: DENY_ALL, ALLOW_ALL, or WHITELIST.
                tenant_id: Used as the namespace and policy name prefix.
                egress_whitelist: List of CIDR strings allowed in WHITELIST mode.

            Returns:
                A dict representing a Kubernetes NetworkPolicy manifest.
            """
            policy: dict[str, Any] = {
                "apiVersion": "networking.k8s.io/v1",
                "kind": "NetworkPolicy",
                "metadata": {
                    "name": f"{tenant_id}-network-policy",
                    "namespace": tenant_id,
                },
                "spec": {
                    "podSelector": {},
                    "policyTypes": ["Ingress", "Egress"],
                },
            }

            if policy_type == NetworkPolicyType.DENY_ALL:
                # Block everything including DNS (port 53)
                policy["spec"]["ingress"] = []
                policy["spec"]["egress"] = []

            elif policy_type == NetworkPolicyType.ALLOW_ALL:
                # Allow all — no DNS restriction in permissive mode
                policy["spec"]["ingress"] = [{}]
                policy["spec"]["egress"] = [{}]

            elif policy_type == NetworkPolicyType.WHITELIST:
                # Block all ingress
                policy["spec"]["ingress"] = []
                # Allow only whitelisted CIDRs, but always block port 53
                allowed_egress: list[dict[str, Any]] = []
                for cidr in (egress_whitelist or []):
                    allowed_egress.append({
                        "to": [{"ipBlock": {"cidr": cidr}}],
                        # No port restriction on whitelisted CIDRs (caller controls)
                    })
                # Explicit deny for port 53 (DNS tunneling prevention)
                # In K8s NetworkPolicy, absence of a rule = deny, so we simply
                # do NOT add port 53 to any allow rule. We document this explicitly.
                policy["spec"]["egress"] = allowed_egress
                policy["spec"]["_dns_tunneling_prevention"] = (
                    "port 53 UDP/TCP not included in any egress allow rule"
                )

            return policy

        def add_egress_whitelist(
            self,
            policy: dict[str, Any],
            cidr: str,
        ) -> dict[str, Any]:
            """Add a CIDR to the egress allow-list of an existing WHITELIST policy.

            Returns a new policy dict (does not mutate the input).
            """
            import copy
            updated = copy.deepcopy(policy)
            egress: list[dict[str, Any]] = updated["spec"].get("egress", [])
            egress.append({"to": [{"ipBlock": {"cidr": cidr}}]})
            updated["spec"]["egress"] = egress
            return updated

        # ------------------------------------------------------------------
        # Boundary crossing log
        # ------------------------------------------------------------------

        def log_boundary_crossing(
            self,
            tenant_id: str,
            direction: str,
            remote_ip: str,
            port: int,
        ) -> NetworkBoundaryCrossing:
            """Record a network boundary crossing event.

            Args:
                tenant_id: The tenant whose boundary was crossed.
                direction: "ingress" or "egress".
                remote_ip: The remote IP address involved.
                port: The port number involved.

            Returns:
                The recorded NetworkBoundaryCrossing.
            """
            crossing = NetworkBoundaryCrossing(
                tenant_id=tenant_id,
                direction=direction,
                remote_ip=remote_ip,
                port=port,
            )
            self._boundary_crossings.append(crossing)
            return crossing

        def get_boundary_crossings(
            self, tenant_id: str | None = None
        ) -> list[NetworkBoundaryCrossing]:
            """Return all logged boundary crossings, optionally filtered by tenant_id."""
            if tenant_id is None:
                return list(self._boundary_crossings)
            return [c for c in self._boundary_crossings if c.tenant_id == tenant_id]

        # ------------------------------------------------------------------
        # Config loading and validation
        # ------------------------------------------------------------------

        def load_config(self, path: str) -> WireGuardConfig:
            """Parse a wgconfig-style .conf file."""
            wg = wgconfig.WGConfig(path)
            wg.read_file()

            iface = wg.interface
            private_key = iface.get("PrivateKey", "")
            address = iface.get("Address", "")
            dns_raw = iface.get("DNS", "")
            dns_list = [d.strip() for d in dns_raw.split(",")] if dns_raw else []

            peers = []
            for peer_pk, peer_data in wg.peers.items():
                allowed_ips_raw = peer_data.get("AllowedIPs", "")
                allowed_ips = [ip.strip() for ip in allowed_ips_raw.split(",")] if allowed_ips_raw else []
                endpoint = peer_data.get("Endpoint")
                pka = peer_data.get("PersistentKeepalive")
                peers.append(
                    PeerConfig(
                        public_key=peer_pk,
                        allowed_ips=allowed_ips,
                        endpoint=endpoint,
                        persistent_keepalive=int(pka) if pka else None,
                    )
                )

            return WireGuardConfig(
                interface_name="wg0",
                private_key=private_key,
                address=address,
                dns=dns_list,
                peers=peers,
            )

        def validate_config(self, config: WireGuardConfig) -> list[str]:
            """Return a list of validation errors for a WireGuardConfig."""
            errors: list[str] = []
            if not config.private_key:
                errors.append("Private key is required")
            if not config.address:
                errors.append("Address is required")
            for peer in config.peers:
                if not peer.public_key:
                    errors.append("Peer public key is required")
                if not peer.allowed_ips:
                    errors.append(f"Peer {peer.public_key} requires AllowedIPs")
            return errors

        # ------------------------------------------------------------------
        # Legacy interface (backward compatibility)
        # ------------------------------------------------------------------

        def apply_policy(
            self, policy_type: NetworkPolicyType, tenant_id: str
        ) -> dict[str, Any]:
            """Legacy alias for generate_network_policy. Prefer generate_network_policy."""
            return self.generate_network_policy(policy_type, tenant_id)
    ```
